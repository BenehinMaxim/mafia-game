// script.js - полная версия с сохранением состояния и кнопкой сброса // ==================== ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ ==================== let players = []; let currentPhase = 'night'; // 'night' или 'day' let logMessages = []; let nightActions = { mafiaKill: null, commissionerCheck: null, doctorSave: null }; // Элементы DOM const playersListEl = document.getElementById('players-list'); const logEl = document.getElementById('log'); const phaseTitleEl = document.getElementById('phase-title'); const nightActionsDiv = document.getElementById('night-actions'); const voteDiv = document.getElementById('vote'); // Ключ для localStorage const STORAGE_KEY = 'mafia_game_state'; // ==================== СОХРАНЕНИЕ И ЗАГРУЗКА ==================== function saveGameState() { const state = { players: players, currentPhase: currentPhase, logMessages: logMessages, nightActions: nightActions, timestamp: new Date().toISOString() }; localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); console.log('✅ Игра сохранена'); } function loadGameState() { const saved = localStorage.getItem(STORAGE_KEY); if (!saved) return false; try { const state = JSON.parse(saved); players = state.players; currentPhase = state.currentPhase; logMessages = state.logMessages; nightActions = state.nightActions || { mafiaKill: null, commissionerCheck: null, doctorSave: null }; // Обновляем интерфейс renderPlayersList(); renderLog(); updatePhaseUI(); return true; } catch (e) { console.error('Ошибка загрузки:', e); return false; } } function resetGame() { if (confirm('⚠️ Вы уверены? Все данные игры будут потеряны!')) { // Очищаем localStorage localStorage.removeItem(STORAGE_KEY); // Сбрасываем переменные players = []; currentPhase = 'night'; logMessages = []; nightActions = { mafiaKill: null, commissionerCheck: null, doctorSave: null }; // Обновляем UI renderPlayersList(); renderLog(); updatePhaseUI(); addLogMessage('🔄 Игра сброшена. Создайте новых игроков.'); console.log('🔄 Игра сброшена'); } } // ==================== ЛОГ ==================== function addLogMessage(message) { const timestamp = new Date().toLocaleTimeString('ru-RU', { hour: '2-digit', minute: '2-digit' }); logMessages.unshift(`[${timestamp}] ${message}`); // Ограничиваем лог 100 сообщениями if (logMessages.length > 100) logMessages.pop(); renderLog(); saveGameState(); } function renderLog() { if (!logEl) return; logEl.innerHTML = logMessages.map(msg => `
${msg}
`).join(''); } // ==================== ИГРОКИ ==================== function addPlayer(name) { if (!name || name.trim() === '') { alert('Введите имя игрока'); return; } if (players.find(p => p.name === name)) { alert('Игрок с таким именем уже существует'); return; } players.push({ id: Date.now() + Math.random(), name: name, role: null, isAlive: true, isLeft: false, voteTarget: null }); renderPlayersList(); addLogMessage(`👤 Добавлен игрок: ${name}`); saveGameState(); } function removePlayer(playerId) { const player = players.find(p => p.id === playerId); if (player && confirm(`Удалить игрока ${player.name}?`)) { players = players.filter(p => p.id !== playerId); renderPlayersList(); addLogMessage(`❌ Удалён игрок: ${player.name}`); saveGameState(); } } function setRole(playerId, role) { const player = players.find(p => p.id === playerId); if (!player) return; player.role = role; renderPlayersList(); addLogMessage(`🎭 Игрок ${player.name} получил роль: ${getRoleName(role)}`); saveGameState(); } function getRoleName(role) { const roles = { 'mafia': '🔪 Мафия', 'commissioner': '🔍 Комиссар', 'doctor': '💊 Доктор', 'civilian': '👨🌾 Мирный', 'left': '🚪 Отошёл' }; return roles[role] || role; } function toggleAlive(playerId) { const player = players.find(p => p.id === playerId); if (!player) return; player.isAlive = !player.isAlive; if (!player.isAlive) { addLogMessage(`💀 Игрок ${player.name} убит`); } else { addLogMessage(`✨ Игрок ${player.name} воскрешён`); } renderPlayersList(); saveGameState(); } // ==================== ФАЗЫ ИГРЫ ==================== function setPhase(phase) { currentPhase = phase; updatePhaseUI(); addLogMessage(phase === 'night' ? '🌙 Наступила ночь' : '☀️ Наступил день'); saveGameState(); } function updatePhaseUI() { if (phaseTitleEl) { phaseTitleEl.textContent = currentPhase === 'night' ? '🌙 Ночь' : '☀️ День'; } if (nightActionsDiv) { nightActionsDiv.style.display = currentPhase === 'night' ? 'block' : 'none'; } if (voteDiv) { voteDiv.style.display = currentPhase === 'day' ? 'block' : 'none'; } } // ==================== НОЧНЫЕ ДЕЙСТВИЯ ==================== function setMafiaKill(playerId) { if (currentPhase !== 'night') return; const player = players.find(p => p.id === playerId); if (!player || !player.isAlive) return; nightActions.mafiaKill = playerId; addLogMessage(`🔪 Мафия выбрала жертву: ${player.name}`); renderNightActionsUI(); saveGameState(); } function setCommissionerCheck(playerId) { if (currentPhase !== 'night') return; const player = players.find(p => p.id === playerId); if (!player || !player.isAlive) return; nightActions.commissionerCheck = playerId; const role = player.role === 'mafia' ? 'МАФИЯ' : 'мирный'; addLogMessage(`🔍 Комиссар проверил ${player.name} — это ${role}`); renderNightActionsUI(); saveGameState(); } function setDoctorSave(playerId) { if (currentPhase !== 'night') return; const player = players.find(p => p.id === playerId); if (!player || !player.isAlive) return; nightActions.doctorSave = playerId; addLogMessage(`💊 Доктор лечит: ${player.name}`); renderNightActionsUI(); saveGameState(); } function resolveNight() { if (currentPhase !== 'night') return; const killedPlayer = players.find(p => p.id === nightActions.mafiaKill); const savedPlayer = players.find(p => p.id === nightActions.doctorSave); if (killedPlayer && killedPlayer.id !== savedPlayer?.id) { killedPlayer.isAlive = false; addLogMessage(`💀 ${killedPlayer.name} был убит мафией!`); } else if (killedPlayer && savedPlayer && killedPlayer.id === savedPlayer.id) { addLogMessage(`💊 Доктор спас ${killedPlayer.name} от смерти!`); } // Сбрасываем ночные действия nightActions = { mafiaKill: null, commissionerCheck: null, doctorSave: null }; setPhase('day'); renderPlayersList(); renderNightActionsUI(); saveGameState(); } function renderNightActionsUI() { const container = document.getElementById('night-actions-ui'); if (!container) return; const alivePlayers = players.filter(p => p.isAlive); container.innerHTML = `

🔪 Мафия (выберите жертву)

🔍 Комиссар (проверить игрока)

💊 Доктор (кого лечить)

`; } // ==================== ОТРИСОВКА ИГРОКОВ ==================== function renderPlayersList() { if (!playersListEl) return; if (players.length === 0) { playersListEl.innerHTML = '
Нет игроков. Добавьте игроков выше.
'; return; } playersListEl.innerHTML = players.map(player => `
${player.name} ${player.role ? getRoleName(player.role) : '⚡ Без роли'} ${player.isAlive ? '❤️ Жив' : '💀 Мёртв'}
`).join(''); } // ==================== ДОБАВЛЕНИЕ КНОПКИ СБРОСА В HTML ==================== function addResetButton() { // Ищем контейнер для кнопок управления const controlPanel = document.querySelector('.control-panel') || document.getElementById('game-controls'); if (controlPanel) { const resetBtn = document.createElement('button'); resetBtn.id = 'reset-game-btn'; resetBtn.textContent = '🔄 Сбросить всё'; resetBtn.className = 'reset-btn'; resetBtn.onclick = resetGame; resetBtn.style.marginLeft = '10px'; resetBtn.style.padding = '8px 16px'; resetBtn.style.backgroundColor = '#dc2626'; resetBtn.style.color = 'white'; resetBtn.style.border = 'none'; resetBtn.style.borderRadius = '6px'; resetBtn.style.cursor = 'pointer'; controlPanel.appendChild(resetBtn); } } // ==================== ИНИЦИАЛИЗАЦИЯ ==================== function init() { // Загружаем сохранённое состояние или создаём новое const loaded = loadGameState(); if (!loaded) { addLogMessage('🎭 Приложение готово! Создайте игроков.'); } else { addLogMessage('📀 Загружено сохранённое состояние игры'); } // Добавляем кнопку сброса addResetButton(); // Настраиваем обработчики для добавления игроков const addPlayerBtn = document.getElementById('add-player-btn'); const playerNameInput = document.getElementById('player-name-input'); if (addPlayerBtn && playerNameInput) { addPlayerBtn.onclick = () => { addPlayer(playerNameInput.value); playerNameInput.value = ''; }; playerNameInput.onkeypress = (e) => { if (e.key === 'Enter') { addPlayer(playerNameInput.value); playerNameInput.value = ''; } }; } // Кнопка смены фазы const togglePhaseBtn = document.getElementById('toggle-phase-btn'); if (togglePhaseBtn) { togglePhaseBtn.onclick = () => { setPhase(currentPhase === 'night' ? 'day' : 'night'); }; } // Рендерим UI ночных действий renderNightActionsUI(); } // Запускаем при загрузке страницы document.addEventListener('DOMContentLoaded', init); // Делаем функции глобальными для доступа из HTML window.addPlayer = addPlayer; window.removePlayer = removePlayer; window.setRole = setRole; window.toggleAlive = toggleAlive; window.setPhase = setPhase; window.setMafiaKill = setMafiaKill; window.setCommissionerCheck = setCommissionerCheck; window.setDoctorSave = setDoctorSave; window.resolveNight = resolveNight; window.resetGame = resetGame;
